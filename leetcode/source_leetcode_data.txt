Merge Intervals
https://leetcode.com/problems/merge-intervals/
class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        if len(intervals) <= 1:
            return intervals
        intervals.sort()
        result = [intervals[0]]
        for s in intervals[1:]:
            if s[0]<=result[-1][1]:
                result[-1][1] = max(result[-1][1], s[1])
            else:
                result.append(s)
        return result
